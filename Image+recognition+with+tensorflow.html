Image Recognition with Tensorflow¶

This notebook shows how to perform image recognition with Tensorflow. We run our classifiers on the MNIST corpus, which contains 28x28 pixel handwritten digits. The corpus is split into 55,000 records for training , 5,000 for validation and 10,000 for testing.

We will organize this notebook as follows. Section 1 imports some libraries and prepare the environment. Section 2 constructs a linear classifer (Multinomial Logistic Regression / Softmax) with a vanilla Gradient Descent Optimizer and perform classification over the MNIST corpus. Section 3 still classifies the MNIST corpus, but this time with a fully-connected deep neural network and Adam Optimizer. We will later see how deep neural network improves the accruacy for image recognition.

def plot_variants(number, numfigs):
    j = 0
    c = 10 # columns of plots
    r = ceil(float(numfigs) / c)
    n = r * c
    plt.rcParams["figure.figsize"] = (10,10)
    fig, ax = plt.subplots(r, c)
    #fig.set_size_inches(r*1.5, c*1.5)
    gs1 = gridspec.GridSpec(r, c)
    gs1.update(wspace=0.05, hspace=0.025) # set the spacing between axes.

    for i in range(len(mnist.test.images)):
        if j >= numfigs:
            if j < n:
                ax[j].axis('off')
                j += 1
                continue
            else:
                break
        predicted, actual = predict(i)
        predicted = int(predicted)
        if predicted == number:
            ax1 = plt.subplot(gs1[j])
            ax1.imshow(mnist.test.images[i].reshape((28,28)), cmap=pylab.cm.gray_r)
            ax1.axis('off')
            ax1.set_aspect('equal')
            ax1.set_title(i)   # set img index as figure title
            j += 1
     
    plt.show()

  • Use Dropout in the second layer to prevent overfitting

# Function to create a fully connected layer
# Initialize the variables that have an affect on how quickly training converges
def fc_layer(input, size_out, name="fc", activation=None):
    with tf.name_scope(name):
        size_in = int(input.shape[1])
        
        # Choose a strategy for weights
        w = tf.Variable(tf.truncated_normal([size_in, size_out], stddev=0.1), name="weights")
        #w = tf.Variable(tf.truncated_normal(shape=[size_in, size_out], stddev=1.0 / sqrt(float(size_in))))
        
        b = tf.Variable(tf.constant(0.1, shape=[size_out]), name="bias")
        wx_plus_b = tf.matmul(input, w) + b
        if activation: return activation(wx_plus_b)
        return wx_plus_b

# Define the model

# First, we'll create two fully connected layers, with ReLU activations
fc1 = fc_layer(images, HIDDEN1_SIZE, "fc1", activation=tf.nn.relu)
fc2 = fc_layer(fc1, HIDDEN2_SIZE, "fc2", activation=tf.nn.relu)

# Next, we'll apply Dropout to the second layer
# This is optional but can help prevent overfitting
dropped = tf.nn.dropout(fc2, keep_prob=0.9)

# Finally, we'll calculate logists. This will be
# the input to our Softmax function. Notice we 
# don't apply an activation at this layer.
# If you've commented out the dropout layer,
# switch the input here to 'fc2'.
y = fc_layer(dropped, NUM_CLASSES, name="output")

# Define loss and an optimizer
with tf.name_scope("loss"):
    loss = tf.reduce_mean(tf.nn.softmax_cross_entropy_with_logits(logits=y, labels=labels))
    tf.summary.scalar('loss', loss)

with tf.name_scope("optimizer"):
    # Whereas in the previous section we used a vanilla GradientDescentOptimizer
    # here, we're using Adam. This is a single line of code change, and more
    # importantly, TensorFlow will still automatically analyze our graph
    # and determine how to adjust the variables to decrease the loss.
    train = tf.train.AdamOptimizer(LEARNING_RATE).minimize(loss)
